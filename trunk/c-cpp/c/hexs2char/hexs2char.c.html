 33 int readword(char *pch, FILE *fpin)
 34 {
 35         int len = 0;
 36         int s;
 37         int flag = 1;
 38         while ((s = fread(pch, sizeof(char), 1, fpin)) == 1)
 39         {
 40                 if (*pch == '\n')
 41                         break;
 42                 else if (isspace(*pch))
 43                 {
 44                         if (flag == 2)
 45                                 break;
 46                         continue;
 47                 }
 48                 flag = 2;
 49                 pch++;
 50                 len++;
 51         }
 52         return *pch == '\n' ? len+1 : len;
 53 }

 55 int readline(char *pch, FILE *fpin)
 56 {
 57         int len = 0;
 58         int s;
 59         while ((s = readword(pch, fpin)) > 0)
 60         {
 61                 pch += s;
 62                 len += s;
 63                 if (*(pch - 1) == '\n')
 64                         break;
 65                 *(pch++) = ' ';
 66                 len++;
 67         }
 68         *pch = '\0';
 69         return len;
 70 }

 72 int transline(char *pline)
 73 {
 74         assert(pline != NULL);
 75         char *ppre,*pascii,*pch = pline;
 76         int chnum, len;
 77 
 78         len = strlen(pline);
 79         while (isspace(*(pch + len -1))) /* 去除行未空格 */
 80                 len--;
 81         while (isspace(*pch)) /* 跳过行前空格 */
 82                 pch++;
 83         while (!isspace(*pch)) /* 跳过第一个单词 */
 84                 pch++;
 85 
 86         pascii = pline + len + 1;
 87         *(pline + len) = '\t';
 88         while (pch < pline + len)
 89         {
 90                 if (pascii == pline + MAX_LINE - 1)
 91                         break;
 92                 while (isspace(*pch))
 93                         pch++;
 94                 ppre = pch;
 95                 while (!isspace(*pch) && pch <= pline + len)
 96                         pch++;
 97                 chnum = hex2dec(ppre, pch - ppre);
 98                 if (32 <= chnum && chnum <= 126)
 99                         *pascii++ = (char)chnum;
100                 else
101                         *pascii++ = (char)(0x82);
102         }
103         *pascii = '\n';
104         *(pascii + 1) = '\0';
105 
106         return 0;
107 }

109 int hex2dec(const char *pbg, int len)
110 {
111         assert(pbg != NULL && len > 0);
112         int pows;
113         int res = 0;
114         char cn[1];
115         const char *p = pbg;
116         while (len > 0)
117         {
118                 pows = (int)powl(16, len - 1);
119                 cn[0] = *p;
120                 if (cn[0] >= '0' && cn[0] <= '9')
121                         res += atoi(cn) * pows;
122                 else if ((cn[0] >= 'a' && cn[0] <= 'f') || (cn[0] >= 'A' && cn[0] <= 'F'))
123                 {
124                         if (cn[0] == 'a' || cn[0] == 'A')
125                                 res += 10 * pows;
126                         if (cn[0] == 'b' || cn[0] == 'B')
127                                 res += 11 * pows;
128                         if (cn[0] == 'c' || cn[0] == 'C')
129                                 res += 12 * pows;
130                         if (cn[0] == 'd' || cn[0] == 'D')
131                                 res += 13 * pows;
132                         if (cn[0] == 'e' || cn[0] == 'E')
133                                 res += 14 * pows;
134                         if (cn[0] == 'f' || cn[0] == 'F')
135                                 res += 15 * pows;
136                 }
137                 else
138                         return 0; /* 无效字符 */
139                 p++;
140                 len--;
141         }
142         return res;
143 }
